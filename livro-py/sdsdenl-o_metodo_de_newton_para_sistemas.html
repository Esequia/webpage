def F(x):  
    y = np.zeros(2)  
 
    y[0] = x[0]**2 - np.cos(x[0]*x[1]) - 1  
    y[1] = np.sin(x[1]) - 2*np.cos(x[0])  
 
    return y  
 
def JF(x):  
    y = np.zeros((2,2))  
 
    y[0,0] = 2*x[0] + x[1]*np.sin(x[0]*x[1])  
    y[0,1] = x[0]*np.sin(x[0]*x[1])  
 
    y[1,0] =  2*np.sin(x[0])  
    y[1,1] = np.cos(x[1])  
 
    return y
