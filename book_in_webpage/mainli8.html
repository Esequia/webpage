function x=q_Jacobi()  
    x0=[0 0 0 0 0]’  
 
    i=0  
    controle=0  
    while controle<3 & i<1000  
    i=i+1  
 
    x(1)=1+x0(2)  
    x(2)=(1+x0(3)+x0(1))/2  
    x(3)=(1+x0(2)+x0(4))/(2+epsilon)  
    x(4)=(1+x0(3)+x0(5))/2  
    x(5)=x0(4)-1  
 
    delta=norm(x-x0,2)  
    if delta<1e-6 then  
        controle=controle+1

        controle=0  
    end  
    mprintf(’i=%d, x1=%f, x5=%f, tol=%.12f\n’,i,x(1),x(5),delta)  
    x0=x;  
    end  
 
endfunction
